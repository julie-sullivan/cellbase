public class Psimi {
    private String psimi;
    private String name;

    public Psimi() {
    }

    public Psimi(String psimi, String name) {
        this.psimi = psimi;
        this.name = name;
    }

    public String getPsimi() {
        return psimi;
    }

    public void setPsimi(String psimi) {
        this.psimi = psimi;
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }
}
